# ============================================================
# System Packages (Ubuntu 22.04 / 24.04)
# ============================================================

# Required:
# sudo apt install -y \
#     build-essential \
#     clang \
#     cmake \
#     git \
#     ninja-build \
#     python3 \
#     python3-pip \
#     python3-venv \
#     libz-dev \
#     libxml2-dev

# Optional (Ubuntu 24.04+ for faster builds):
# sudo apt install libprotobuf-dev protobuf-compiler

# ============================================================
# LLVM/MLIR (Must be built from source)
# ============================================================

# Location: ~/dev/llvm-project/
# Version: >= 18.0 (21.0+ recommended)
# Build time: ~30-90 minutes
# Disk space: ~20-30 GB

# Build commands:
# git clone https://github.com/llvm/llvm-project.git ~/dev/llvm-project
# cd ~/dev/llvm-project && mkdir build && cd build
# cmake -G Ninja ../llvm \
#     -DLLVM_ENABLE_PROJECTS="mlir" \
#     -DLLVM_TARGETS_TO_BUILD="X86;NVPTX;AMDGPU" \
#     -DLLVM_ENABLE_ASSERTIONS=ON \
#     -DCMAKE_BUILD_TYPE=Release \
#     -DCMAKE_C_COMPILER=clang \
#     -DCMAKE_CXX_COMPILER=clang++
# ninja -j$(nproc)

# ============================================================
# Dependencies Managed by CMake FetchContent
# ============================================================

# The following are automatically downloaded and built:
# - Protocol Buffers 29.2.0 (if not using system protobuf)
# - pybind11_protobuf (main branch)
# - pybind11 (via pybind11_protobuf)
# - abseil (via pybind11_protobuf)

# ============================================================
# Version Requirements
# ============================================================

# CMake: >= 3.20
# LLVM/MLIR: >= 18.0
# Python: >= 3.8
# Clang: >= 14
# Protobuf: >= 3.20 (or auto-fetched 29.2.0)
# C++ Standard: C++17

# ============================================================
# Build Process
# ============================================================

# 1. Install system packages (above)
# 2. Build LLVM/MLIR (above)
# 3. python3 -m venv venv && source venv/bin/activate
# 4. pip install -e ".[dev]"
# 5. ./build.sh

# First build: 2-15 minutes (depends on protobuf)
# Incremental builds: 15-30 seconds
